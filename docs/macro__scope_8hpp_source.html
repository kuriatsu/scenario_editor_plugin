98 #define NLOHMANN_JSON_SERIALIZE_ENUM(ENUM_TYPE, ...) \
99  template<typename BasicJsonType> \
100  inline void to_json(BasicJsonType& j, const ENUM_TYPE& e) \
101  { \
102  static_assert(std::is_enum<ENUM_TYPE>::value, #ENUM_TYPE " must be an enum!"); \
103  static const std::pair<ENUM_TYPE, BasicJsonType> m[] = __VA_ARGS__; \
104  auto it = std::find_if(std::begin(m), std::end(m), \
105  [e](const std::pair<ENUM_TYPE, BasicJsonType>& ej_pair) -> bool \
106  { \
107  return ej_pair.first == e; \
108  }); \
109  j = ((it != std::end(m)) ? it : std::begin(m))->second; \
110  } \
111  template<typename BasicJsonType> \
112  inline void from_json(const BasicJsonType& j, ENUM_TYPE& e) \
113  { \
114  static_assert(std::is_enum<ENUM_TYPE>::value, #ENUM_TYPE " must be an enum!"); \
115  static const std::pair<ENUM_TYPE, BasicJsonType> m[] = __VA_ARGS__; \
116  auto it = std::find_if(std::begin(m), std::end(m), \
117  [&j](const std::pair<ENUM_TYPE, BasicJsonType>& ej_pair) -> bool \
118  { \
119  return ej_pair.second == j; \
120  }); \
121  e = ((it != std::end(m)) ? it : std::begin(m))->first; \
122  }
